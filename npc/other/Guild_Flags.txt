function	script	Guild_Register	{
	if( set(.@Guild_ID, getcharid(2)) == 0 )
	{
		mes "[Registration Flag]";
		mes "Only members of autorized Guilds can be converted into their Jobs or check the Flag information.";
	}
	else if( getvariableofnpc(.Guild_ID,getarg(0)) == 0 && Autorized == 0 )
	{
		if( getguildmaster(.@Guild_ID) == strcharinfo(0) )
		{
			if( countitem(7773) < 1 )
			{
				mes "[Registration Flag]";
				mes "You don't have a ^0000FFRegistration Medal^000000. It's required to register your guild.";
				mes "More information on Kafra Staff.";
				close;
			}

			// NPC Data
			// ***********************************************************
			flagemblem .@Guild_ID,getarg(0);
			flagemblem .@Guild_ID,"GuildRoom" + getarg(1);
			flagemblem .@Guild_ID,"GuildHouse" + getarg(1);
			set getvariableofnpc(.Guild_ID,getarg(0)), .@Guild_ID;
			setnpcdisplay getarg(0), getguildname(.@Guild_ID), 722;

			// Guild Data
			// ***********************************************************
			if( $@Guild_Lvl )
				atcommand "@guildlevel " + $@Guild_Lvl;
			
			set getvariableofnpc(.Members,getarg(0)), $@MaxGuildMembers; // Control of Available Slots
			set getvariableofnpc(.Converted,getarg(0)), 0; // Count of Converted Members
			set getvariableofnpc(.ItemCount,getarg(0)), 0; // Count of Received Items (to control abuse)
			set getvariableofnpc(.Weapons,getarg(0)), 0;

			if( $@Guild_Zeny > 0 )
				set Zeny, Zeny + $@Guild_Zeny;
			set .@k, getarraysize($@LimitPerParty);
			for( set .@i, 0; .@i < .@k; set .@i, .@i + 1 )
				set getvariableofnpc(.Jobs[.@i],getarg(0)), $@LimitPerParty[.@i] * $@PartyPerGuild;
			
			// Ammo Settings
			set .@k, getarraysize($@Item);
			for( set .@i, 0; .@i < .@k; set .@i, .@i + 1 )
				set getvariableofnpc(.Ammo[.@i],"GuildAmmo" + getarg(1)), 0;

			announce "[" + strcharinfo(0) + "] has registered the Guild [" + getguildname(.@Guild_ID) + "]",0,0x00CCFF;
			misceffect 537;
			misceffect 259;

			// Character Data
			// ***********************************************************
			set Autorized, 1;
			save "rentinb" + getarg(1), 31, 12;

			mes "[Registration Flag]";
			mes "Congratulations, you have registered your Guild in this Tournament.";
			mes "Now, each guild member and you, talk to me to choose your Class.";
			next;
			mes "[Registration Flag]";
			mes "You Guild has received " + $@Guild_Lvl + " levels to allow you to configure all Guild Skills and invite the rest of the Members.";
			mes "Each Registered and Converted member will increase your inventory for the Event.";
		}
		else
		{
			mes "[Registration Flag]";
			mes "Only the Guild Master can register their Guild in a free flag.";
		}
	}
	else if( .@Guild_ID == getvariableofnpc(.Guild_ID,getarg(0)) )
	{
		if( Converted )
		{
			if( Guild_ID != .@Guild_ID )
			{
				mes "[Registration Flag]";
				mes "This is not your original registration flag. Sorry, return to your original guild.";
				close;
			}

			mes "[Registration Flag]";
			mes "Do you want to enter to the Guild House?";
			next;
			if( getguildmaster(.@Guild_ID) == strcharinfo(0) )
				set .@Menu, select("Yes, warp me inside:No, thanks:^0000FFGuild Leader Options^000000");
			else
				set .@Menu, select("Yes, warp me inside:No, thanks");
			
			if( .@Menu == 2 )
			{ // Cancel
				mes "[Registration Flag]";
				mes "Enjoy the Battle.";
				close;
			}
			else if( .@Menu == 3 )
			{
				mes "[Registration Flag]";
				mes "This is the Guild Leader Menu.";
				mes "What do you want to do?";
				next;

				switch( select( "Request Special Weapons","Transfer my Guild Leader position","Remove Guild Members" ) )
				{
					case 1:
						mes "[Registration Flag]";
						mes "To receive the Guild Weapons, at least 50% of the total of invited players per guild need to be already registered and transformed into their classes.";
						next;
						if( $@Guild_Weapons == 0 )
						{
							mes "[Registration Flag]";
							mes "I am sorry, this event don't allow Guild Weapons.";
							close;
						}
						if( getvariableofnpc(.Weapons,getarg(0)) != 0 )
						{
							mes "[Registration Flag]";
							mes "Your guild has already received the Guild Weapons.";
							close;
						}

						if( getvariableofnpc(.Converted,getarg(0)) < ($@PartyPerGuild * 6) )
						{
							mes "[Registration Flag]";
							mes "Your Guild has already registered " + getvariableofnpc(.Converted,getarg(0)) + " members.";
							mes "You need at least " + ($@PartyPerGuild * 6) + " members to receive the Weapons.";
							close;
						}

						getitem2 1228,1,1,4,0,0,0,0,0; // Combat Knife
						getitem2 1224,1,1,4,0,0,0,0,0; // Sword Breaker
						getitem2 1225,1,1,4,0,0,0,0,0; // Mail Breaker
						getitem2 1230,1,1,4,0,0,0,0,0; // Ice Pick
						set getvariableofnpc(.Weapons,getarg(0)), 1;
						mes "[Registration Flag]";
						mes "Here are your Guild Weapons. Only this equipment can be traded with other players, so choose who inside your Guild are going to use it.";
						break;
					case 2:
						mes "[Registration Flag]";
						mes "Who will be the new Guild Leader?";
						mes "This player need to be Online.";
						next;

						set .@Members$, "";
						for( set .@i, 0; getvariableofnpc(.Char_ID[.@i],getarg(0)) != 0 && .@i < 127; set .@i, .@i + 1 )
							set .@Members$, .@Members$ + getvariableofnpc(.Char_Name$[.@i],getarg(0)) + ":";

						set .@Menu, select(.@Members$) - 1;
						set .@NewGL$, getvariableofnpc(.Char_Name$[.@Menu],getarg(0));

						if( strcharinfo(0) == .@NewGL$ )
						{
							mes "[Registration Flag]";
							mes "No changes where done.";
							close;
						}

						mes "[Registration Flag]";
						mes "You are going to transfer your Guild Leader position to ^0000FF" + .@NewGL$ + "^000000.";
						next;

						if( select("Proceed:Cancel") == 2 )
						{
							mes "[Registration Flag]";
							mes "No changes where done.";
							close;
						}

						if( agitcheck() )
						{
							mes "[Registration Flag]";
							mes "You cannot transfer the Guild on WoE time.";
							close;
						}

						atcommand "@changegm " + .@NewGL$ + "";
						mes "[Registration Flag]";
						mes "The change has been requested to the Server.";
						mes "If there is a mistake, maybe the destination player is not online.";
						mes "If the changes where done, please, both players, you and the new Guild Leader, restart your games (logout/login).";
						break;
					case 3:
						mes "[Registration Flag]";
						mes "This is the list of your registered Guild Members.";
						mes "Please choose the member you are going to remove.";
						next;
						mes "[Registration Flag]";
						mes "WARNING:";
						mes "- The player will be banned from the server.";
						mes "- A new account will be required if he wants to join again.";
						mes "- The character name will not be available anymore.";
						mes "- Remember to take any usable/healing item the user have in their account first.";
						next;
						set .@Menu$, "";
						for( set .@i, 0; getvariableofnpc(.Char_ID[.@i],getarg(0)) != 0 && .@i < 127; set .@i, .@i + 1 )
						{
							set .@j, getvariableofnpc(.Char_Job[.@i],getarg(0));
							set .@Menu$, .@Menu$ + getvariableofnpc(.Char_Name$[.@i],getarg(0)) + " [" + jobname($@Classes[.@j]) + "]:";
						}

						set .@Index, select(.@Menu$) - 1;
						set .@Name$, getvariableofnpc(.Char_Name$[.@Index],getarg(0));
						set .@Job, getvariableofnpc(.Char_Job[.@Index],getarg(0));
						
						mes "[Registration Flag]";
						mes "Do you really want to remove the selected member?";
						next;
						if( select("Remove it:Cancel") == 2 )
						{
							mes "[Registration Flag]";
							mes "No changes where done...";
							close;
						}

						if( .@Name$ == strcharinfo(0) )
						{
							mes "[Registration Flag]";
							mes "You cannot remove yourself from the Guild.";
							close;
						}
						
						atcommand "@block " + .@Name$ + "";
						set getvariableofnpc(.Members,getarg(0)), getvariableofnpc(.Members,getarg(0)) + 1;
						set getvariableofnpc(.Converted,getarg(0)), getvariableofnpc(.Converted,getarg(0)) - 1;
						set getvariableofnpc(.Jobs[.@Job],getarg(0)), getvariableofnpc(.Jobs[.@Job],getarg(0)) + 1;

						for( set .@i, .@Index; getvariableofnpc(.Char_ID[.@i],getarg(0)) != 0; set .@i, .@i + 1 )
						{
							set getvariableofnpc(.Char_ID[.@i],getarg(0)), getvariableofnpc(.Char_ID[.@i + 1],getarg(0));
							set getvariableofnpc(.Char_Name$[.@i],getarg(0)), getvariableofnpc(.Char_Name$[.@i + 1],getarg(0));
							set getvariableofnpc(.Char_Job[.@i],getarg(0)), getvariableofnpc(.Char_Job[.@i + 1],getarg(0));
						}
						
						mes "[Registration Flag]";
						mes "The user has been removed from the guild.";
						mes "The slot is available again (for members and class)";
						break;
				}
				close;
			}

			mes "[Registration Flag]";
			mes "Close this conversation to be warped into the House.";
			close2;
			warp "rentb" + getarg(1) + "",24,6;
			end;
		}
		else
		{
			mes "[Registration Flag]";
			mes "Are you ready to be transformed into the Class you are going to use in the Event?.";
			next;
			set .@Job, select($@Menu$) - 1;

			mes "[Registration Flag]";
			mes "You choose ^0000FF" + jobname($@Classes[.@Job]) + "^000000.";
			mes "Please, think about it. Next button will be available in 10 seconds.";
			sleep2 10000;
			mes "Are you sure about your selection?";
			next;
			if( select("^0000FFYes, i am sure^000000:No, sorry... I will choose another Class.") == 2 )
			{
				mes "[Registration Flag]";
				mes "No changes done.";
				close;
			}
			if( getvariableofnpc(.Jobs[.@Job],getarg(0)) < 1 )
			{
				mes "[Registration Flag]";
				mes "There are no more free slots for this Class.";
				close;
			}
			if( ($@Classes[.@Job] == Job_Bard || $@Classes[.@Job] == Job_Clown || $@Classes[.@Job] == Job_Baby_Bard) && Sex == 0 )
			{
				mes "[Registration Flag]";
				mes "The selected class is not available for your gender.";
				close;
			}
			if( ($@Classes[.@Job] == Job_Dancer || $@Classes[.@Job] == Job_Gypsy || $@Classes[.@Job] == Job_Baby_Dancer) && Sex == 1 )
			{
				mes "[Registration Flag]";
				mes "The selected class is not available for your gender.";
				close;
			}
			if( getvariableofnpc(.Members,getarg(0)) < 1 )
			{
				mes "[Registration Flag]";
				mes "There are no more free slots for the Guild.";
				close;
			}

			callfunc "convert_job",$@Classes[.@Job];
			if( Converted )
			{
				if( !Autorized )
				{
					set Autorized, 1;
					save "rentinb" + getarg(1), 31, 12;
					announce "[" + strcharinfo(0) + "] has joined the Guild [" + getguildname(.@Guild_ID) + "]",0,0xCC9900;
				}

				mes "[Registration Flag]";
				mes "How you are ready. Talk to me again if you want to enter the Guild House.";
				
				// Data Update
				set getvariableofnpc(.Jobs[.@Job],getarg(0)), getvariableofnpc(.Jobs[.@Job],getarg(0)) - 1;
				set getvariableofnpc(.Members,getarg(0)), getvariableofnpc(.Members,getarg(0)) - 1;
				set getvariableofnpc(.Converted,getarg(0)), getvariableofnpc(.Converted,getarg(0)) + 1;

				// Increase Consumible Items
				if( getvariableofnpc(.ItemCount,getarg(0)) < getvariableofnpc(.Converted,getarg(0)) )
				{
					set getvariableofnpc(.ItemCount,getarg(0)), getvariableofnpc(.ItemCount,getarg(0)) + 1;

					set .@k, getarraysize($@Item);
					for( set .@i, 0; .@i < .@k; set .@i, .@i + 1 )
						set getvariableofnpc(.Ammo[.@i],"GuildAmmo" + getarg(1)), getvariableofnpc(.Ammo[.@i],"GuildAmmo" + getarg(1)) + (($@Ammo[.@i] * 2) / 1);
				}

				set Guild_ID, .@Guild_ID; // Stores Guild ID
				
				for( set .@i, 0; getvariableofnpc(.Char_ID[.@i],getarg(0)) != 0 && .@i < 127; set .@i, .@i + 1 )
					; // Search a Free Slot
				if( .@i < 127 )
				{ // Store Char Info
					set getvariableofnpc(.Char_ID[.@i],getarg(0)), getcharid(0);
					set getvariableofnpc(.Char_Name$[.@i],getarg(0)), strcharinfo(0);
					set getvariableofnpc(.Char_Job[.@i],getarg(0)), .@Job; // Index to job.
				}
			}
			else
			{
				mes "[Registration Flag]";
				mes "There was a problem while trying to convert you into the selected Class. Please try again.";
			}
		}
	}
	else
	{
		if( getvariableofnpc(.Guild_ID,getarg(0)) )
		{
			mes "^FFA500* " + getguildname(getvariableofnpc(.Guild_ID,getarg(0))) + " *^000000";
			for( set .@i, 0; getvariableofnpc(.Char_ID[.@i],getarg(0)) != 0 && .@i < 127; set .@i, .@i + 1 )
			{ // Guild Members Info
				set .@j, getvariableofnpc(.Char_Job[.@i],getarg(0));
				mes "" + getvariableofnpc(.Char_Name$[.@i],getarg(0)) + "";
				mes "^FFFFFF_____^000000[^0000FF" + jobname($@Classes[.@j]) + "^000000]";
			}

			if( getgmlevel() > 90 )
			{
				next;
				mes "[Registration Flag]";
				mes "Game Master, deseas accesar la Casa Privada de la guild?";
				next;
				if( select("Si, enviame ahora:No Gracias") == 2 )
				{
					mes "[Registration Flag]";
					mes "Suerte con la organización.";
					close;
				}

				mes "[Registration Flag]";
				mes "Cierra esta conversación para ser enviado.";
				close2;
				warp "rentb" + getarg(1) + "",24,6;
				end;
			}
		}
		else
		{
			mes "[Registration Flag]";
			mes "This is not your Guild Flag.";
			mes "Your guild is already registered in another flag.";
		}
	}
	close;
}
